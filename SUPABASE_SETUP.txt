-- 1. Create Profiles Table (Updated for Public Access & Donation Date)
create table if not exists public.profiles (
  id uuid references auth.users on delete cascade primary key,
  full_name text,
  blood_group text,
  phone text,
  location text,
  address text,
  status text default 'Available', -- Available / Unavailable
  role text default 'user',
  donations_count int default 0,
  last_donation_date date, -- ৩ মাসের হিসাবের জন্য
  bio text,
  avatar_url text,
  updated_at timestamp with time zone default now()
);

-- 2. Create Blood Requests Table
create table if not exists public.blood_requests (
  id uuid default gen_random_uuid() primary key,
  patient_name text not null,
  blood_group text not null,
  location text not null,
  phone text not null,
  posted_by text default 'Guest', 
  status text default 'pending', 
  accepted_by text,
  created_at timestamp with time zone default now()
);

-- 3. Storage Setup (Run these in SQL Editor)
insert into storage.buckets (id, name, public) 
values ('avatars', 'avatars', true)
on conflict (id) do nothing;

-- 4. POWERFUL PUBLIC POLICIES (No Login Required)
alter table public.profiles enable row level security;
alter table public.blood_requests enable row level security;

-- Allow ANYONE to view everything
create policy "Public View Profiles" on public.profiles for select using (true);
create policy "Public View Requests" on public.blood_requests for select using (true);

-- Allow ANYONE to insert requests (Public Request)
create policy "Public Insert Requests" on public.blood_requests for insert with check (true);

-- Allow Users to manage their own profile (If they decide to login/join)
create policy "Users Manage Own Profile" on public.profiles for all using (auth.uid() = id);

-- Storage Policies
drop policy if exists "Public Access" on storage.objects;
create policy "Public Access" on storage.objects for select using (bucket_id = 'avatars');
create policy "Public Upload" on storage.objects for insert with check (bucket_id = 'avatars');
